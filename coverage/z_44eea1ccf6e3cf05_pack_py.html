2"""tokenpak/pack.py — High-level ContextPack API with compile reports. 

3 

4Provides the developer-facing compile() API with full transparency: 

5 

6 from tokenpak.pack import ContextPack, PackBlock 

7 

8 pack = ContextPack(budget=8000) 

9 pack.add(PackBlock(id="system_prompt", type="instructions", content="...", priority="critical")) 

10 pack.add(PackBlock(id="api_docs", type="knowledge", content="...", priority="high", max_tokens=1000)) 

11 pack.add(PackBlock(id="search_003", type="evidence", content="...", priority="medium", quality=0.3)) 

12 pack.add(PackBlock(id="history", type="conversation", content="...", priority="low", max_tokens=650)) 

13 

14 compiled = pack.compile() 

15 print(compiled.report) # human-readable terminal report 

16 print(compiled.text) # final wire-format text 

17 print(compiled.report.to_json()) # Langfuse-ready dict 

18 

19Compile logic (in priority order): 

20 1. CRITICAL blocks: always KEPT (never removed/truncated unless block-level max_tokens set) 

21 2. HIGH blocks: KEPT, truncated if block exceeds max_tokens 

22 3. MEDIUM blocks: KEPT if budget allows; quality-filtered first (removed if quality < threshold) 

23 4. LOW blocks: KEPT last, truncated to fill remaining budget; dropped if no room 

24 

25Quality filtering (for MEDIUM/LOW): blocks with quality < quality_threshold are REMOVED. 

26Block-level max_tokens: blocks exceeding their own cap are COMPACTED (truncated within cap). 

27Budget overflow: lowest-priority blocks REMOVED until within budget. 

28"""

45 def _count_tokens(text: str) -> int:

48 def _truncate_to_tokens(text: str, max_tokens: int) -> str: 

49 tokens = _enc.encode(text) 

50 if len(tokens) <= max_tokens: 

51 return text 

52 return _enc.decode(tokens[:max_tokens]) + "..."

56 def _count_tokens(text: str) -> int: 

57 return max(1, len(text) // 4)

59 def _truncate_to_tokens(text: str, max_tokens: int) -> str: 

60 approx = max_tokens * 4 

61 return text[:approx] + ("..." if len(text) > approx else "")

251class ContextPack: 

252 """Budget-aware context compiler with full transparency reports. 

253 

254 Args: 

255 budget: Total token budget for the compiled output. 

256 quality_threshold: Blocks with quality < this are REMOVED (default 0.5). 

257 separator: String placed between blocks in text output. 

258 """ 

259 

260 def __init__( 

261 self, 

262 budget: int = 8000, 

263 quality_threshold: float = 0.5, 

264 separator: str = "\n\n---\n\n", 

265 ) -> None: 

266 self.budget = budget 

267 self.quality_threshold = quality_threshold 

268 self.separator = separator 

269 self._blocks: List[PackBlock] = [] 

270 

271 def add(self, block: PackBlock) -> "ContextPack": 

272 """Add a block. Returns self for chaining.""" 

273 self._blocks.append(block) 

274 return self 

275 

276 def clear(self) -> "ContextPack": 

277 """Remove all blocks.""" 

278 self._blocks.clear() 

279 return self

281 # ── Compile ─────────────────────────────────────────────────────────── 

282 

283 def compile(self) -> CompiledResult: 

284 """Compile all blocks into a budgeted output with a full report. 

285 

286 Returns: 

287 CompiledResult with .text and .report attributes. 

288 """ 

289 t_start = time.perf_counter() 

290 

291 # Measure raw token counts 

292 block_tokens: dict[str, int] = {b.id: _count_tokens(b.content) for b in self._blocks} 

293 input_tokens_total = sum(block_tokens.values()) 

294 

295 decisions: list[Decision] = [] 

296 kept_blocks: list[tuple[PackBlock, str, int]] = [] # (block, text, tokens) 

297 

298 # Sort by priority (critical first) for deterministic ordering 

299 ordered = sorted(self._blocks, key=lambda b: _priority_rank(b.priority)) 

300 

301 for block in ordered: 

302 raw_tokens = block_tokens[block.id] 

303 

304 # ── Quality filter (REMOVED) ────────────────────────────── 

305 if block.quality is not None and block.quality < self.quality_threshold: 

306 decisions.append( 

307 Decision( 

308 block_id=block.id, 

309 block_type=block.type, 

310 action=Action.REMOVED, 

311 reason=( 

312 f"below quality threshold " 

313 f"({block.quality:.2f} < {self.quality_threshold:.2f})" 

314 ), 

315 priority=block.priority, 

316 tokens_before=raw_tokens, 

317 tokens_after=0, 

318 quality=block.quality, 

319 ) 

320 ) 

321 continue 

322 

323 text = block.content 

324 

325 # ── Block-level cap (COMPACTED) ─────────────────────────── 

326 if block.max_tokens is not None and raw_tokens > block.max_tokens: 

327 text = _truncate_to_tokens(text, block.max_tokens) 

328 after_tokens = _count_tokens(text) 

329 decisions.append( 

330 Decision( 

331 block_id=block.id, 

332 block_type=block.type, 

333 action=Action.COMPACTED, 

334 reason=f"exceeded block budget (max {block.max_tokens:,})", 

335 method="extractive_truncation", 

336 priority=block.priority, 

337 tokens_before=raw_tokens, 

338 tokens_after=after_tokens, 

339 ) 

340 ) 

341 kept_blocks.append((block, text, after_tokens)) 

342 continue 

343 

344 # ── Tentative KEPT ──────────────────────────────────────── 

345 kept_blocks.append((block, text, raw_tokens)) 

346 

347 # ── Budget enforcement pass ─────────────────────────────────── 

348 # Remove lowest-priority blocks until we fit within budget. 

349 # For blocks already in decisions (REMOVED/COMPACTED), they're done. 

350 # For kept_blocks, we may need to drop the lowest-priority ones. 

351 sum(t for _, _, t in kept_blocks) 

352 

353 # Sort kept by priority DESCENDING (lowest priority last → drop first) 

354 kept_by_prio = sorted( 

355 kept_blocks, 

356 key=lambda x: _priority_rank(x[0].priority), 

357 reverse=True, # highest rank number = lowest priority = drop first 

358 ) 

359 

360 final_kept: list[tuple[PackBlock, str, int]] = [] 

361 running_total = 0 

362 

363 # We pass through from highest to lowest priority 

364 # Flip to process highest-priority first 

365 for block, text, tokens in reversed(kept_by_prio): 

366 if running_total + tokens <= self.budget: 

367 final_kept.append((block, text, tokens)) 

368 running_total += tokens 

369 else: 

370 # Can we TRUNCATE to fit remaining budget? 

371 remaining = self.budget - running_total 

372 if remaining > 0 and block.priority in ("critical", "high"): 

373 truncated = _truncate_to_tokens(text, remaining) 

374 after_tokens = _count_tokens(truncated) 

375 final_kept.append((block, truncated, after_tokens)) 

376 running_total += after_tokens 

377 decisions.append( 

378 Decision( 

379 block_id=block.id, 

380 block_type=block.type, 

381 action=Action.TRUNCATED, 

382 reason="conversation budget exceeded", 

383 priority=block.priority, 

384 tokens_before=tokens, 

385 tokens_after=after_tokens, 

386 ) 

387 ) 

388 else: 

389 # REMOVED due to budget 

390 decisions.append( 

391 Decision( 

392 block_id=block.id, 

393 block_type=block.type, 

394 action=Action.REMOVED, 

395 reason="over total budget — dropped lowest priority block", 

396 priority=block.priority, 

397 tokens_before=tokens, 

398 tokens_after=0, 

399 quality=block.quality, 

400 ) 

401 ) 

402 

403 # Blocks that were quietly KEPT (no special action) need decisions too 

404 {b.id for b, _, _ in final_kept} 

405 decision_ids = {d.block_id for d in decisions} 

406 

407 for block, text, tokens in final_kept: 

408 if block.id not in decision_ids: 

409 decisions.append( 

410 Decision( 

411 block_id=block.id, 

412 block_type=block.type, 

413 action=Action.KEPT, 

414 reason=( 

415 "critical priority" 

416 if block.priority == "critical" 

417 else f"{block.priority} priority — within budget" 

418 ), 

419 priority=block.priority, 

420 tokens_before=block_tokens[block.id], 

421 tokens_after=tokens, 

422 quality=block.quality, 

423 ) 

424 ) 

425 

426 # Sort final_kept back into priority order for output 

427 priority_idx = {b.id: _priority_rank(b.priority) for b in self._blocks} 

428 final_sorted = sorted(final_kept, key=lambda x: priority_idx[x[0].id]) 

429 

430 # Sort decisions in evaluation order (priority, then original index) 

431 original_order = {b.id: i for i, b in enumerate(self._blocks)} 

432 decisions.sort(key=lambda d: (original_order.get(d.block_id, 999),)) 

433 

434 # ── Build output text ───────────────────────────────────────── 

435 parts = [text for _, text, _ in final_sorted] 

436 output_text = self.separator.join(parts) 

437 

438 output_tokens = sum(t for _, _, t in final_sorted) 

439 final_order = [b.id for b, _, _ in final_sorted] 

440 

441 t_end = time.perf_counter() 

442 compile_time_ms = (t_end - t_start) * 1000.0 

443 

444 report = CompileReport( 

445 input_blocks=len(self._blocks), 

446 output_blocks=len(final_sorted), 

447 input_tokens=input_tokens_total, 

448 output_tokens=output_tokens, 

449 budget=self.budget, 

450 compile_time_ms=compile_time_ms, 

451 decisions=decisions, 

452 final_order=final_order, 

453 ) 

454 

455 return CompiledResult(text=output_text, report=report)
