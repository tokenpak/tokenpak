20 def normalize(self, body: bytes) -> CanonicalRequest: 

21 data = json.loads(body) 

22 

23 consumed = {"model", "system", "messages", "tools", "stream"} 

24 generation: Dict[str, Any] = {} 

25 raw_extra: Dict[str, Any] = {} 

26 

27 for key, value in data.items(): 

28 if key in consumed: 

29 continue 

30 if key in {"max_tokens", "temperature", "top_p", "top_k", "stop_sequences", "metadata"}: 

31 generation[key] = value 

32 else: 

33 raw_extra[key] = value 

34 

35 return CanonicalRequest( 

36 model=data.get("model", "unknown"), 

37 system=copy.deepcopy(data.get("system", "")), 

38 messages=copy.deepcopy(data.get("messages", [])), 

39 tools=copy.deepcopy(data.get("tools")), 

40 generation=generation, 

41 stream=bool(data.get("stream", False)), 

42 raw_extra=raw_extra, 

43 source_format=self.source_format, 

44 )

46 def denormalize(self, canonical: CanonicalRequest) -> bytes: 

47 payload: Dict[str, Any] = { 

48 "model": canonical.model, 

49 "messages": copy.deepcopy(canonical.messages), 

50 "stream": canonical.stream,

52 if canonical.system not in (None, "", []):

61 def inject_system_context(self, body: bytes, injection_text: str) -> bytes: 

62 """ 

63 Inject volatile content into the system prompt with correct cache boundary. 

64 

65 Cache boundary strategy (Anthropic prompt caching): 

66 - STABLE prefix (original system blocks) → gets cache_control: ephemeral 

67 - VOLATILE injection (retrieval/vault content) → NO cache_control 

68 

69 Placing cache_control on the volatile block causes cache churn (0% hit rate) 

70 because the volatile content changes every request. The cache_control must 

71 anchor the STABLE prefix so Anthropic can cache up to that boundary. 

72 """ 

73 canonical = self.normalize(body) 

74 # Volatile injection block — intentionally NO cache_control 

75 volatile_block: dict = {"type": "text", "text": injection_text} 

76 

77 if isinstance(canonical.system, str): 

78 if canonical.system: 

79 # String system → convert to list; stable block gets the cache marker 

80 canonical.system = [ 

81 { 

82 "type": "text", 

83 "text": canonical.system, 

84 "cache_control": {"type": "ephemeral"}, 

85 }, 

86 volatile_block, # volatile: no cache_control 

87 ] 

88 else: 

89 # Empty system — just store raw text (no stable prefix to mark) 

90 canonical.system = injection_text 

91 elif isinstance(canonical.system, list): 

92 # Mark the LAST EXISTING block (stable prefix anchor) 

93 if canonical.system: 

94 marked = list(canonical.system) 

95 for i in range(len(marked) - 1, -1, -1): 

96 blk = marked[i] 

97 if isinstance(blk, dict) and blk.get("type") == "text": 

98 if blk.get("cache_control") != {"type": "ephemeral"}: 

99 marked[i] = dict(blk, cache_control={"type": "ephemeral"}) 

100 break 

101 canonical.system = marked 

102 canonical.system.append(volatile_block) # volatile: no cache_control 

103 else: 

104 canonical.system = injection_text 

105 return self.denormalize(canonical)
