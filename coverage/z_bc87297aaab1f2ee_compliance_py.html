1"""Enterprise Compliance Reports — SOC2, GDPR, CCPA. 

2 

3Generates structured compliance reports by pulling data from the audit log 

4and system configuration. 

5 

6Usage:: 

7 

8 from tokenpak.enterprise.compliance import ComplianceReporter 

9 

10 reporter = ComplianceReporter(audit_db=".tokenpak/audit.db") 

11 report = reporter.generate("soc2") 

12 print(report.as_text()) 

13 report.save("soc2-report-2026-Q1.json") 

14"""

31@dataclass 

32class ComplianceControl: 

33 """A single compliance control with status and evidence.""" 

34 

35 id: str 

36 name: str 

37 description: str 

38 status: str # "compliant" | "partial" | "not_applicable" | "gap" 

39 evidence: list[str] = field(default_factory=list) 

40 notes: str = "" 

41 

42 

43@dataclass 

44class ComplianceReport: 

45 """Full compliance report for a given standard.""" 

46 

47 standard: str # "soc2" | "gdpr" | "ccpa" 

48 generated_at: str 

49 period_start: str 

50 period_end: str 

51 organization: str 

52 controls: list[ComplianceControl] = field(default_factory=list) 

53 summary: dict = field(default_factory=dict) 

54 raw_audit_stats: dict = field(default_factory=dict) 

55 

56 def as_dict(self) -> dict: 

57 return { 

58 "standard": self.standard, 

59 "generated_at": self.generated_at, 

60 "period_start": self.period_start, 

61 "period_end": self.period_end, 

62 "organization": self.organization, 

63 "summary": self.summary, 

64 "raw_audit_stats": self.raw_audit_stats, 

65 "controls": [asdict(c) for c in self.controls], 

66 } 

67 

68 def as_json(self, indent: int = 2) -> str: 

69 return json.dumps(self.as_dict(), indent=indent) 

70 

71 def as_text(self) -> str: 

72 lines = [ 

73 f"{'=' * 64}", 

74 f"COMPLIANCE REPORT: {self.standard.upper()}", 

75 f"Generated: {self.generated_at}", 

76 f"Period: {self.period_start} → {self.period_end}", 

77 f"Org: {self.organization}", 

78 f"{'=' * 64}", 

79 "", 

80 ] 

81 # Summary 

82 s = self.summary 

83 compliant = s.get("compliant", 0) 

84 total = s.get("total_controls", len(self.controls)) 

85 pct = round(compliant / total * 100) if total else 0 

86 lines += [ 

87 f"SUMMARY: {compliant}/{total} controls compliant ({pct}%)", 

88 f" Gaps: {s.get('gap', 0)}", 

89 f" Partial: {s.get('partial', 0)}", 

90 f" N/A: {s.get('not_applicable', 0)}", 

91 "", 

92 "CONTROLS:", 

93 ] 

94 icon_map = {"compliant": "✅", "partial": "⚠️", "gap": "❌", "not_applicable": "➖"} 

95 for ctrl in self.controls: 

96 icon = icon_map.get(ctrl.status, "?") 

97 lines.append(f" {icon} [{ctrl.id}] {ctrl.name}") 

98 if ctrl.status != "compliant": 

99 lines.append(f" Status: {ctrl.status}") 

100 if ctrl.notes: 

101 lines.append(f" Note: {ctrl.notes}") 

102 for ev in ctrl.evidence: 

103 lines.append(f" Evidence: {ev}") 

104 lines.append("") 

105 return "\n".join(lines) 

106 

107 def save(self, path: Union[str, Path], fmt: str = "json") -> None: 

108 out = Path(path) 

109 out.parent.mkdir(parents=True, exist_ok=True) 

110 if fmt == "json": 

111 out.write_text(self.as_json(), encoding="utf-8") 

112 elif fmt == "text": 

113 out.write_text(self.as_text(), encoding="utf-8") 

114 else: 

115 raise ValueError(f"Unsupported format: {fmt!r}")

123class ComplianceReporter: 

124 """Generate compliance reports from audit log + config. 

125 

126 Parameters 

127 ---------- 

128 audit_db: 

129 Path to the audit SQLite database. 

130 organization: 

131 Organization name to include in reports. 

132 """ 

133 

134 _STANDARDS = {"soc2", "gdpr", "ccpa"} 

135 

136 def __init__( 

137 self, 

138 audit_db: Union[str, Path, None] = None, 

139 organization: str = "Your Organization", 

140 ) -> None: 

141 self._audit_path = str(audit_db or default_audit_path()) 

142 self.organization = organization 

143 

144 def generate( 

145 self, 

146 standard: str, 

147 since: Optional[str] = None, 

148 until: Optional[str] = None, 

149 ) -> ComplianceReport: 

150 """Generate a compliance report for *standard*. 

151 

152 Parameters 

153 ---------- 

154 standard: 

155 One of ``'soc2'``, ``'gdpr'``, ``'ccpa'``. 

156 since, until: 

157 ISO date strings to bound the audit data window. 

158 """ 

159 standard = standard.lower() 

160 if standard not in self._STANDARDS: 

161 raise ValueError( 

162 f"Unknown compliance standard: {standard!r}. " 

163 f"Choose from: {sorted(self._STANDARDS)}" 

164 ) 

165 

166 now = datetime.now(tz=timezone.utc) 

167 period_end = until or now.strftime("%Y-%m-%d") 

168 period_start = since or ( 

169 datetime.fromtimestamp(now.timestamp() - 90 * 86400, tz=timezone.utc).strftime( 

170 "%Y-%m-%d" 

171 ) 

172 ) 

173 

174 # Pull stats from audit log 

175 audit_stats: dict = {} 

176 try: 

177 with AuditLog(self._audit_path) as log: 

178 audit_stats = log.summary(since=period_start) 

179 except Exception as exc: 

180 audit_stats = {"error": str(exc)} 

181 

182 # Build controls per standard 

183 builder = getattr(self, f"_build_{standard}_controls") 

184 controls = builder(audit_stats) 

185 

186 # Summarise 

187 tally: dict[str, int] = {} 

188 for c in controls: 

189 tally[c.status] = tally.get(c.status, 0) + 1 

190 summary = { 

191 "total_controls": len(controls), 

192 **tally, 

193 } 

194 

195 return ComplianceReport( 

196 standard=standard, 

197 generated_at=now.isoformat(), 

198 period_start=period_start, 

199 period_end=period_end, 

200 organization=self.organization, 

201 controls=controls, 

202 summary=summary, 

203 raw_audit_stats=audit_stats, 

204 )
