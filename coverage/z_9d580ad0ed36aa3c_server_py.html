73def parse_filter(filter_str: Optional[str]) -> dict[str, str]: 

74 """Parse a filter DSL string into a dict of field:value pairs. 

75 

76 Format: ``"provider:anthropic,model:opus,agent:sue"`` 

77 

78 Parameters 

79 ---------- 

80 filter_str: 

81 Comma-separated key:value pairs, or None. 

82 

83 Returns 

84 ------- 

85 dict 

86 Parsed filters. Empty dict if input is None or empty. 

87 """ 

88 if not filter_str: 

89 return {} 

90 result = {} 

91 for part in filter_str.split(","): 

92 part = part.strip() 

93 if ":" in part: 

94 key, value = part.split(":", 1) 

95 key = key.strip().lower() 

96 value = value.strip() 

97 if key and value: 

98 # Normalize key names 

99 if key in ("provider", "model", "agent", "agent_id", "status", "start", "end"): 

100 if key == "agent": 

101 key = "agent_id" 

102 result[key] = value 

103 return result

171 @app.get("/v1/health") 

172 async def health(): 

173 """Rich health check — ingest status, error rate, last event, request counts.""" 

174 import time as _t 

175 

176 try: 

177 cur = _storage._conn.cursor() 

178 now = _t.time() 

179 # Last event timestamp 

180 cur.execute("SELECT MAX(ts) FROM tp_events") 

181 last_ts = cur.fetchone()[0] 

182 last_event_iso = None 

183 if last_ts: 

184 import datetime as _dt 

185 

186 last_event_iso = _dt.datetime.utcfromtimestamp(last_ts).strftime( 

187 "%Y-%m-%dT%H:%M:%SZ" 

188 ) 

189 

190 # Error rate last 1h 

191 cutoff_1h = now - 3600 

192 cur.execute( 

193 "SELECT COUNT(*), SUM(CASE WHEN status='error' THEN 1 ELSE 0 END) FROM tp_events WHERE ts >= ?", 

194 (cutoff_1h,), 

195 ) 

196 row = cur.fetchone() 

197 total_1h = row[0] or 0 

198 errors_1h = row[1] or 0 

199 error_rate_1h = round(errors_1h / total_1h, 4) if total_1h > 0 else 0.0 

200 

201 # Requests last 24h 

202 cur.execute("SELECT COUNT(*) FROM tp_events WHERE ts >= ?", (now - 86400,)) 

203 requests_24h = cur.fetchone()[0] or 0 

204 

205 # Data lag (seconds since last event) 

206 data_lag_s = round(now - last_ts, 0) if last_ts else None 

207 

208 # Overall status 

209 is_stale = data_lag_s is not None and data_lag_s > 300 

210 status = "healthy" 

211 if error_rate_1h > 0.10: 

212 status = "degraded" 

213 if error_rate_1h > 0.50 or (data_lag_s is not None and data_lag_s > 1800): 

214 status = "down" 

215 

216 return { 

217 "status": status, 

218 "service": "tokenpak-telemetry", 

219 "ingest_active": True, 

220 "last_event_ts": last_event_iso, 

221 "last_event_age_s": data_lag_s, 

222 "error_rate_1h": error_rate_1h, 

223 "requests_24h": requests_24h, 

224 "is_stale": is_stale, 

225 } 

226 except Exception as e: 

227 return {"status": "degraded", "service": "tokenpak-telemetry", "error": str(e)}

281 @app.post("/v1/telemetry/ingest", response_model=IngestResponse) 

282 async def ingest(request: IngestRequest, background_tasks: BackgroundTasks): 

283 """Ingest a batch of telemetry events into the local DB.""" 

284 start_time = time.perf_counter() 

285 results, processed, failed = [], 0, 0 

286 for idx, event in enumerate(request.events): 

287 event_start = time.perf_counter() 

288 try: 

289 raw_event = event.model_dump(exclude_none=True) 

290 if event.raw: 

291 raw_event.update(event.raw) 

292 loop = asyncio.get_event_loop() 

293 pipeline_result: PipelineResult = await loop.run_in_executor( 

294 _executor, _pipeline.process, raw_event 

295 ) 

296 duration_ms = (time.perf_counter() - event_start) * 1000 

297 if pipeline_result.success: 

298 processed += 1 

299 results.append( 

300 EventResult( 

301 index=idx, 

302 success=True, 

303 event_id=pipeline_result.event_id, 

304 duration_ms=duration_ms, 

305 ) 

306 ) 

307 else: 

308 failed += 1 

309 results.append( 

310 EventResult( 

311 index=idx, 

312 success=False, 

313 event_id=pipeline_result.event_id, 

314 error=pipeline_result.error, 

315 duration_ms=duration_ms, 

316 partial=pipeline_result.partial_data_stored, 

317 ) 

318 ) 

319 except Exception as e: 

320 failed += 1 

321 results.append( 

322 EventResult( 

323 index=idx, 

324 success=False, 

325 error=str(e), 

326 duration_ms=(time.perf_counter() - event_start) * 1000, 

327 ) 

328 ) 

329 if processed > 0: 

330 _cache.invalidate_prefix("kpi:") 

331 _cache.invalidate_prefix("rollup:") 

332 _cache.invalidate_prefix("filters:") 

333 return IngestResponse( 

334 success=(failed == 0), 

335 total=len(request.events), 

336 processed=processed, 

337 failed=failed, 

338 results=results, 

339 total_duration_ms=(time.perf_counter() - start_time) * 1000, 

340 )
