1""" 

2tokenpak/handlers/rate_limit.py 

3~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ 

4Exponential backoff handler for 429 / rate-limit responses. 

5 

6Supports: 

7- Exponential backoff with configurable base_wait and multiplier (2^attempt) 

8- max_wait ceiling 

9- Retry-After header override (capped at max_wait) 

10- Optional jitter via jitter_factor (additive, stays ≤ max_wait) 

11- Deterministic mode when jitter_factor=0 

12""" 

13 

14from __future__ import annotations 

15 

16import random 

17 

18 

19class RateLimitBackoff: 

20 """Exponential backoff calculator for rate-limited API requests. 

21 

22 Parameters 

23 ---------- 

24 base_wait : float 

25 Base wait time in seconds for attempt 0. Default: 1.0. 

26 max_wait : float 

27 Maximum wait time ceiling in seconds. Default: 60.0. 

28 jitter_factor : float 

29 Fraction of computed wait to add as random jitter (0 = deterministic). 

30 Jitter is drawn from [0, jitter_factor * computed_wait). Default: 0.1. 

31 """ 

32 

33 def __init__( 

34 self, 

35 base_wait: float = 1.0, 

36 max_wait: float = 60.0, 

37 jitter_factor: float = 0.1, 

38 ) -> None: 

39 self.base_wait = base_wait 

40 self.max_wait = max_wait 

41 self.jitter_factor = jitter_factor 

42 

43 def wait_time(self, attempt: int, retry_after: float | None = None) -> float: 

44 """Calculate the wait time for a given attempt number. 

45 

46 Parameters 

47 ---------- 

48 attempt : int 

49 Zero-indexed retry attempt number. Negative values yield sub-base waits. 

50 retry_after : float | None 

51 If provided (e.g. from a ``Retry-After`` header), use this value 

52 directly instead of computing exponential backoff. Still capped at 

53 ``max_wait``. 

54 

55 Returns 

56 ------- 

57 float 

58 Seconds to wait before the next retry. 

59 """ 

60 if retry_after is not None: 

61 return min(retry_after, self.max_wait) 

62 

63 # Exponential: base_wait * 2^attempt 

64 computed = self.base_wait * (2.0 ** attempt) 

65 

66 # Apply jitter (additive, uniform in [0, jitter_factor * computed)) 

67 if self.jitter_factor > 0: 

68 computed += random.random() * self.jitter_factor * computed 

69 

70 # Cap at max_wait 

71 return min(computed, self.max_wait)
