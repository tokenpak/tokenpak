1""" 

2tokenpak/api/routes.py 

3 

4HTTP route definitions for the TokenPak proxy management API. 

5 

6Currently exposes: 

7 GET /health — Liveness + provider + cache status 

8 GET /metrics — Prometheus text format metrics 

9 

10Usage (standalone / testing):: 

11 

12 from tokenpak.api.routes import HealthRoute, MetricsRoute 

13 

14 checker = HealthRoute() # or pass start_time=<float> 

15 payload = checker.handle() # returns dict, HTTP status always 200 

16 

17 metrics = MetricsRoute() 

18 body, status, headers = metrics.handle_bytes() 

19 

20Integration with ProxyServer 

21----------------------------- 

22The ProxyServer (tokenpak/agent/proxy/server.py) already handles GET /health 

23via ``ProxyServer.health()``. This module provides the *new* structured 

24endpoint that matches the documented schema (healthy|degraded|unhealthy + 

25per-provider checks + cache metrics). 

26 

27To wire it in, add to ``_ProxyHandler.do_GET``:: 

28 

29 if path == "/health" or path.startswith("/health?"): 

30 self._send_json(ps._health_route.handle()) 

31 return 

32 

33 if path == "/metrics": 

34 body, status, hdrs = ps._metrics_route.handle_bytes() 

35 ... 

36 

37and initialise in ``ProxyServer.__init__``:: 

38 

39 ps._health_route = HealthRoute(start_time=ps.session['start_time']) 

40 ps._metrics_route = MetricsRoute(proxy_server=ps) 

41"""

52class HealthRoute: 

53 """ 

54 Handles GET /health requests. 

55 

56 Parameters 

57 ---------- 

58 start_time : float, optional 

59 Proxy start time (Unix epoch). Defaults to module import time if not 

60 provided — useful for standalone/test usage. 

61 version : str, optional 

62 Override proxy version string. 

63 """ 

64 

65 def __init__( 

66 self, 

67 start_time: Optional[float] = None, 

68 version: Optional[str] = None, 

69 ) -> None: 

70 self._checker = HealthChecker( 

71 start_time=start_time or time.time(), 

72 version=version, 

73 ) 

74 

75 def handle(self) -> Dict[str, Any]: 

76 """ 

77 Run all health checks and return the response dict. 

78 

79 HTTP response is always 200; status detail is in the JSON body. 

80 """ 

81 return self._checker.check() 

82 

83 def handle_bytes(self) -> Tuple[bytes, int, Dict[str, str]]: 

84 """ 

85 Return ``(body_bytes, http_status, headers)`` for direct HTTP handler use. 

86 

87 status is always 200 — consumers must inspect ``payload["status"]``. 

88 """ 

89 payload = self.handle() 

90 body = json.dumps(payload, indent=2).encode("utf-8") 

91 headers = { 

92 "Content-Type": "application/json", 

93 "Content-Length": str(len(body)), 

94 "Access-Control-Allow-Origin": "*", 

95 "Cache-Control": "no-cache", 

96 } 

97 return body, 200, headers

105class RouteRegistry: 

106 """ 

107 Minimal route registry for management API endpoints. 

108 

109 Supports exact-path matching only (no regex/params). 

110 """ 

111 

112 def __init__(self) -> None: 

113 self._routes: Dict[str, Any] = {} 

114 

115 def register(self, path: str, handler: Any) -> None: 

116 """Register *handler* for *path*.""" 

117 self._routes[path] = handler 

118 

119 def match(self, path: str) -> Optional[Any]: 

120 """Return the handler for *path*, or None if not registered.""" 

121 # Strip query string for matching 

122 clean = path.split("?")[0] 

123 return self._routes.get(clean) 

124 

125 def paths(self) -> list[str]: 

126 return list(self._routes.keys())

129class MetricsRoute: 

130 """ 

131 Handles GET /metrics requests — returns Prometheus text exposition format. 

132 

133 Parameters 

134 ---------- 

135 proxy_server : ProxyServer, optional 

136 Live proxy server instance for session + circuit-breaker data. 

137 If None, metrics are collected from available global registries only. 

138 db_path : str or Path, optional 

139 Path to TelemetryDB for per-provider/model breakdowns. 

140 Defaults to the project-level ``telemetry.db`` when not set. 

141 """ 

142 

143 def __init__( 

144 self, 

145 proxy_server: Optional[Any] = None, 

146 db_path: Optional[str] = None, 

147 ) -> None: 

148 self._proxy_server = proxy_server 

149 self._db_path = db_path 

150 

151 def handle(self) -> str: 

152 """Collect and return Prometheus metrics as a text string.""" 

153 from tokenpak.monitoring.metrics import ProxyMetricsCollector 

154 

155 collector = ProxyMetricsCollector( 

156 proxy_server=self._proxy_server, 

157 db_path=self._db_path, 

158 ) 

159 return collector.collect() 

160 

161 def handle_bytes(self) -> Tuple[bytes, int, Dict[str, str]]: 

162 """ 

163 Return ``(body_bytes, http_status, headers)`` for direct HTTP handler use. 

164 """ 

165 body = self.handle().encode("utf-8") 

166 headers = { 

167 "Content-Type": "text/plain; version=0.0.4; charset=utf-8", 

168 "Content-Length": str(len(body)), 

169 "Access-Control-Allow-Origin": "*", 

170 "Cache-Control": "no-cache", 

171 } 

172 return body, 200, headers

175def build_default_registry(start_time: Optional[float] = None) -> RouteRegistry: 

176 """ 

177 Build and return a RouteRegistry pre-populated with default management routes. 

178 

179 Parameters 

180 ---------- 

181 start_time : float, optional 

182 Passed through to HealthRoute for accurate uptime reporting. 

183 """ 

184 registry = RouteRegistry() 

185 registry.register("/health", HealthRoute(start_time=start_time)) 

186 registry.register("/metrics", MetricsRoute()) 

187 return registry
