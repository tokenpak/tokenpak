72class VaultWatcher: 

73 """Watch directories and trigger re-indexing on file changes. 

74 

75 Features: 

76 - watchdog-based filesystem events (inotify/FSEvents) 

77 - Debounced re-indexing: coalesces rapid bursts into one reindex 

78 - Pattern filtering (ignore __pycache__, .git, etc.) 

79 - Status / stats reporting 

80 - Graceful Ctrl+C handling when blocking=True

96 def start(self, blocking: bool = False) -> None: 

97 """Start watching. If blocking=True, run until Ctrl+C.""" 

98 try: 

99 from watchdog.events import FileSystemEventHandler 

100 from watchdog.observers import Observer 

101 except ImportError: 

102 raise RuntimeError("watchdog is required: pip install watchdog")

104 self._stats = WatcherStats() 

105 self._running = True

108 class _Handler(FileSystemEventHandler): 

109 def on_any_event(self, event): 

110 if event.is_directory: 

111 return 

112 src = getattr(event, "src_path", None) 

113 if src and not watcher_ref._should_ignore(src): 

114 watcher_ref._on_fs_event(src) 

115 

116 self._observer: BaseObserver = Observer() # type: ignore 

117 for raw_path in self.config.watch_paths: 

118 watch_dir = str(Path(raw_path).expanduser().resolve()) 

119 self._observer.schedule(_Handler(), watch_dir, recursive=self.config.recursive) # type: ignore 

120 logger.info("Watching: %s", watch_dir) 

121 

122 self._observer.start() # type: ignore

124 self._debounce_thread = threading.Thread( 

125 target=self._debounce_loop, daemon=True, name="tp-watcher-debounce" 

126 ) 

127 self._debounce_thread.start()

158 def _load_patterns_from_file(self, filepath: str) -> List[str]: 

159 """Read patterns from a .gitignore-style file, skipping comments/blanks.""" 

160 patterns = [] 

161 try: 

162 with open(filepath, "r", encoding="utf-8", errors="ignore") as f: 

163 for line in f: 

164 line = line.rstrip("\n") 

165 stripped = line.strip() 

166 if not stripped or stripped.startswith("#"): 

167 continue 

168 # Negation patterns (!) are not yet supported — skip 

169 if stripped.startswith("!"): 

170 continue 

171 patterns.append(stripped) 

172 except OSError: 

173 pass 

174 return patterns

176 def _load_ignore_files(self) -> None: 

177 """Scan watch paths for .gitignore and .tokenpakignore files.""" 

178 self._gitignore_patterns = [] 

179 self._tokenpakignore_patterns = [] 

180 for raw_path in self.config.watch_paths: 

181 watch_dir = Path(raw_path).expanduser().resolve() 

182 if self.config.use_gitignore: 

183 gi_path = watch_dir / ".gitignore" 

184 self._gitignore_patterns.extend(self._load_patterns_from_file(str(gi_path))) 

185 if self.config.use_tokenpakignore: 

186 tpi_path = watch_dir / ".tokenpakignore" 

187 self._tokenpakignore_patterns.extend(self._load_patterns_from_file(str(tpi_path))) 

188 if self._gitignore_patterns: 

189 logger.info("Loaded %d pattern(s) from .gitignore", len(self._gitignore_patterns)) 

190 if self._tokenpakignore_patterns: 

191 logger.info( 

192 "Loaded %d pattern(s) from .tokenpakignore", len(self._tokenpakignore_patterns)

195 def _should_ignore(self, path: str) -> bool: 

196 """Return True if path matches any ignore pattern (config, .gitignore, .tokenpakignore).""" 

197 p = Path(path) 

198 parts = p.parts 

199 

200 all_patterns = list(self.config.ignore_patterns or []) 

201 all_patterns.extend(self._gitignore_patterns) 

202 all_patterns.extend(self._tokenpakignore_patterns) 

203 

204 for pattern in all_patterns: 

205 # Directory pattern (trailing slash) — match against path parts 

206 if pattern.endswith("/"): 

207 dir_pattern = pattern.rstrip("/") 

208 for part in parts: 

209 if fnmatch.fnmatch(part, dir_pattern): 

210 return True 

211 # Pattern with slash = relative path match 

212 elif "/" in pattern: 

213 # Match against the full path string 

214 if fnmatch.fnmatch(str(p), f"*{pattern}") or fnmatch.fnmatch(str(p), pattern): 

215 return True 

216 else: 

217 # Simple name/extension pattern — match against each path component 

218 for part in parts: 

219 if fnmatch.fnmatch(part, pattern): 

220 return True 

221 return False

223 def _on_fs_event(self, path: str) -> None: 

224 self._stats.events_received += 1 

225 with self._debounce_lock: 

226 self._pending[path] = time.monotonic() 

227 

228 def _debounce_loop(self) -> None: 

229 debounce_s = self.config.debounce_ms / 1000.0 

230 while self._running: 

231 time.sleep(max(debounce_s / 4, 0.05)) 

232 now = time.monotonic() 

233 ready = [] 

234 with self._debounce_lock: 

235 for path, ts in list(self._pending.items()): 

236 if now - ts >= debounce_s: 

237 ready.append(path) 

238 for path in ready: 

239 del self._pending[path] 

240 if ready: 

241 self._reindex(ready)

243 def _reindex(self, changed_paths: list) -> None: 

244 self._stats.reindexes_triggered += 1 

245 reindexed = 0 

246 try: 

247 import hashlib 

248 

249 from tokenpak.processors import get_processor 

250 from tokenpak.registry import BlockRegistry 

251 from tokenpak.tokens import count_tokens 

252 from tokenpak.walker import FILE_TYPES 

253 

254 db = self.config.db_path 

255 registry = BlockRegistry(db) if db else BlockRegistry() 

256 

257 for path in changed_paths: 

258 p = Path(path) 

259 if not p.exists(): 

260 continue 

261 ext = p.suffix.lower() 

262 file_type = FILE_TYPES.get(ext) 

263 if file_type not in ("code", "text", "data"): 

264 continue 

265 try: 

266 content = p.read_text(encoding="utf-8", errors="ignore") 

267 except OSError: 

268 continue 

269 

270 if not registry.has_changed(str(p), content): 

271 continue 

272 

273 processor = get_processor(file_type) 

274 if processor is None: 

275 continue 

276 compressed = processor.process(content, str(p)) 

277 

278 content_hash = hashlib.sha256(content.encode()).hexdigest() 

279 block = { 

280 "id": f"{p}#{content_hash[:8]}", 

281 "path": str(p), 

282 "content_hash": content_hash, 

283 "file_type": file_type, 

284 "raw_tokens": count_tokens(content), 

285 "compressed_tokens": count_tokens(compressed), 

286 "compressed_content": compressed, 

287 "metadata": "{}", 

288 } 

289 registry.add_block(block) # type: ignore 

290 reindexed += 1 

291 logger.debug("Re-indexed: %s", path) 

292 

293 if self.on_change: 

294 try: 

295 self.on_change(path) 

296 except Exception: 

297 pass 

298 

299 except Exception as exc: 

300 logger.error("Reindex error: %s", exc) 

301 return 

302 

303 self._stats.files_reindexed += reindexed 

304 if reindexed: 

305 logger.info( 

306 "Re-indexed %d file(s) | total reindexes=%d files=%d", 

307 reindexed, 

308 self._stats.reindexes_triggered, 

309 self._stats.files_reindexed, 

310 )

312 def _run_blocking(self) -> None: 

313 watched = [str(Path(p).expanduser().resolve()) for p in self.config.watch_paths] 

314 print(f"[tokenpak] Watching {len(watched)} path(s) — press Ctrl+C to stop") 

315 for p in watched: 

316 print(f" → {p}") 

317 print(f" debounce: {self.config.debounce_ms}ms | recursive: {self.config.recursive}") 

318 print()
